func ChooseVideo(videos []Video, time int) (Video, error) { chosen := Video{ ID: 0, Duration: 0, Entertainment: 0, } confirma := false for i := range videos { if videos[i].Duration <= time && videos[i].Entertainment > chosen.Entertainment { chosen = videos[i] confirma = true } time-- } if confirma { return chosen, nil } return chosen, fmt.Errorf("Não há videos") }
